find_package(PkgConfig)
pkg_check_modules(DBUS dbus-1)

include_directories(
  ${DBUS_INCLUDE_DIRS}
  ${BOOST_INCLUDE_DIRS}
)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=c++20 -Wall -fno-strict-aliasing -fvisibility=hidden -fvisibility-inlines-hidden -Wextra -fPIC")

add_library(
  ${PROJECT_NAME} SHARED
  plugin.cc
  data.cc
  application.cc
  config.cc
  define.pb.cc
  log.cc
  utils.cc
  writer.cc
  resp_chain.cc
  define.cc
)

target_link_libraries (
  ${PROJECT_NAME}

  ${BOOST_LIBRARIES}
  ${DBUS_CPP_LIBRARIES}
  ${CRYPTOPP_LIBRARIES}
  ${CPR_LIBRARIES}
  ${SQLITE3_LIBRARIES}
  ${PROTOBUF_LIBRARIES}
)

install(TARGET ${PLUGIN_NAME} LIBRARY DESTINATION lib/dde-plugins/session)
